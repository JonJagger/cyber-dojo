def enter
  avatar = kata.start_avatar    
  full = avatar.nil?
  #one_self.started(avatar) if !full
  render json: {
    avatar_name: !full ? avatar.name : nil,
    full: full,
    enter_dialog_html: !full ? enter_dialog_html(avatar.name) : '',
    full_dialog_html: full ? full_dialog_html : ''
  }
end
